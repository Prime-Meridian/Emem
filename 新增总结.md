        // 【新增】构建包含全局记忆的长期记忆上下文的通用函数
        function buildLongTermMemoryContext(currentChat, currentChatId) {
            let longTermMemoryContext = '';
            const currentMemories = [];
            
            // 添加当前角色的长期记忆
            if (currentChat.longTermMemory && currentChat.longTermMemory.length > 0) {
                currentMemories.push(...currentChat.longTermMemory.map(mem => `- ${mem.content}`));
            }
            
            // 添加全局长期记忆（来自其他角色）
            const globalMemories = [];
            for (const [chatId, otherChat] of Object.entries(chats)) {
                if (chatId !== currentChatId && 
                    otherChat.globalMemoryEnabled && 
                    otherChat.longTermMemory && 
                    otherChat.longTermMemory.length > 0) {
                    const otherChatName = otherChat.originalName || otherChat.name || '未知角色';
                    globalMemories.push(`## 来自 ${otherChatName} 的全局记忆`);
                    globalMemories.push(...otherChat.longTermMemory.map(mem => `- ${mem.content}`));
                }
            }
            
            // 构建完整的长期记忆上下文
            if (currentMemories.length > 0 || globalMemories.length > 0) {
                const memoryParts = [];
                
                if (currentMemories.length > 0) {
                    memoryParts.push(`# 你的长期记忆 (已发生的事实)\n${currentMemories.join('\n')}`);
                }
                
                if (globalMemories.length > 0) {
                    memoryParts.push(`\n# 全局长期记忆 (其他角色的重要记忆)\n${globalMemories.join('\n')}`);
                }
                
                longTermMemoryContext = memoryParts.join('\n');
            }
            
            return longTermMemoryContext;
        }

        /**

         */
        function getJailbreakSuffix() {
            const jailbreakEnabled = localStorage.getItem('jailbreakEnabled') === 'true';
            if (!jailbreakEnabled) return '';
            
            const jailbreakText = localStorage.getItem('jailbreakText') || '';
            if (!jailbreakText.trim()) return '';
            
            return '\n\n' + jailbreakText.trim();
        }
        
        /**
         * 【核心】检查并触发自动总结
         * @param {string} chatId 
         */
        async function checkAndTriggerAutoSummary(chatId) {
            const chat = state.chats[chatId];
            if (!chat || !chat.settings.enableAutoMemory) return;
        
            const lastSummaryTimestamp = chat.lastMemorySummaryTimestamp || 0;
            const messagesSinceLastSummary = chat.history.filter(m => m.timestamp > lastSummaryTimestamp && !m.isHidden);
        
            if (messagesSinceLastSummary.length >= chat.settings.autoMemoryInterval) {
                console.log(`达到自动总结阈值 (${messagesSinceLastSummary.length}/${chat.settings.autoMemoryInterval})，开始总结...`);
                await triggerAutoSummary(chatId);
            }
        }
